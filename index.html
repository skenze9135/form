ðŸ”§ WiFi Connection Troubleshooting

Common WiFi Connection Fixes:

  1. 1. Check SSID & Password: Make sure they're correct in your ESP32 code
  2. 2. Signal Strength: Move ESP32 closer to router (needs RSSI > -75 dBm)
  3. 3. 2.4GHz Only: ESP32 doesn't support 5GHz networks
  4. 4. Special Characters: Avoid @ # $ % in WiFi password
  5. 5. Hidden SSID: Make network visible or use WiFi.begin(ssid, pass, channel, bssid)

Updated ESP32 Code (With Error Handling):

// Better WiFi connection with retry logic
const char* WIFI_SSID = "YourHotspot";     // Must be 2.4GHz
const char* WIFI_PASS = "YourPassword";    // Exact password
const char* POST_URL  = "http://192.168.1.100/api/wifi_report"; // Use IP not HTTPS

void connectWiFi() {
  WiFi.mode(WIFI_STA);  // Station mode
  WiFi.disconnect();    // Clear old config
  delay(100);
  
  lcd.clear();
  lcd.print("Connecting WiFi");
  
  WiFi.begin(WIFI_SSID, WIFI_PASS);
  
  int attempts = 0;
  while(WiFi.status() != WL_CONNECTED && attempts < 30) {
    delay(500);
    lcd.setCursor(attempts % 16, 1);
    lcd.print(".");
    attempts++;
    
    // Show debug info
    Serial.print(".");
    Serial.print(" Status: ");
    Serial.println(WiFi.status());
  }
  
  if(WiFi.status() == WL_CONNECTED) {
    lcd.clear();
    lcd.print("WiFi OK!");
    lcd.setCursor(0,1);
    lcd.print(WiFi.localIP());
    Serial.println("Connected! IP: " + WiFi.localIP().toString());
  } else {
    lcd.clear();
    lcd.print("WiFi FAIL!");
    lcd.setCursor(0,1);
    
    // Show error code
    switch(WiFi.status()) {
      case WL_NO_SSID_AVAIL:
        lcd.print("SSID not found");
        Serial.println("Error: SSID not found");
        break;
      case WL_CONNECT_FAILED:
        lcd.print("Wrong password");
        Serial.println("Error: Wrong password");
        break;
      case WL_CONNECTION_LOST:
        lcd.print("Lost connection");
        Serial.println("Error: Connection lost");
        break;
      default:
        lcd.print("Error: " + String(WiFi.status()));
        Serial.println("Error code: " + String(WiFi.status()));
    }
    
    // Send error to website
    sendError(WiFi.status());
  }
}

void sendError(int errorCode) {
  // If somehow connected, send error details
  if(WiFi.status() == WL_CONNECTED) {
    HTTPClient http;
    http.begin(POST_URL);
    http.addHeader("Content-Type", "application/json");
    
    String json = "{\"error\":\"WiFi Error " + String(errorCode) + "\",";
    json += "\"ssid\":\"" + String(WIFI_SSID) + "\"}";
    
    http.POST(json);
    http.end();
  }
}
